\begin{dfn}{}{tag}
	1以上$n$以下の整数の集合を$[n]$で表す、
	つまり
	\begin{align}
		[n] = \{ 1, 2, \dots, n \}
	\end{align}
\end{dfn}

\begin{dfn}{集合の濃度が等しい}{濃度の等しさ}
	集合$A$と$B$の濃度が等しいとは、
	集合$A$から$B$への全単射が存在することである。
\end{dfn}
